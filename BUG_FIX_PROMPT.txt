```python
def _monitor_positions(self):
    """Poll active positions every 15s, call _execute_exit on resolution."""
    import time as _time
    log.info("[Monitor] Position monitor started")
    while not self._stop_event.is_set():
        try:
            with self._state_lock:
                open_ids = list(self._active_positions.keys())

            for mid in open_ids:
                try:
                    with self._state_lock:
                        pos = self._active_positions.get(mid)
                    if pos is None:
                        continue

                    slug = getattr(pos, 'slug', None) or mid
                    resolved, winner = self._check_market_resolved(slug)

                    if resolved:
                        our_side   = getattr(pos, 'side', 'YES')
                        exit_price = 1.0 if (winner or '').upper() == our_side.upper() else 0.0
                        log.info(
                            f"[Monitor] {mid} RESOLVED — winner={winner} "
                            f"our={our_side} exit_price={exit_price}"
                        )
                        self._execute_exit(pos, exit_price, ExitReason.RESOLVED)
                    else:
                        # Time fallback: force-close if way past expiry
                        age = _time.time() - getattr(pos, 'entry_time', _time.time())
                        tf  = int(getattr(pos, 'tf', 5))
                        if age > (tf * 60) + 120:  # tf + 2 min buffer
                            log.warning(f"[Monitor] {mid} age={age:.0f}s forcing exit")
                            # BUG IS HERE - what's being passed to _execute_exit?
                            self._execute_exit(pos, 0.5, ExitReason.TIMEOUT)

                except Exception as e:
                    log.warning(f"[Monitor] Error checking {mid}: {e}")

            _time.sleep(15)
        except Exception as e:
            log.error(f"[Monitor] Loop error: {e}", exc_info=True)
            _time.sleep(15)
```

```python
def record_exit(self, trade_id: str, exit_price: float, exit_reason: str = ''):

        rec = self._trades.pop(trade_id, None)
        # [FIX-3B] Fuzzy match added
        if rec is None:

                (k, v) for k, v in self._trades.items()

                best_key = max(candidates, key=lambda x: x[1].entry_time)[0]
```
